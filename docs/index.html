Install

Basic Usage

import { DateTimePicker } from 'crisli-picker';

function MyComponent() {
  const [date, setDate] = useState(new Date());

  return (
    <DateTimePicker
      value={date}
      onChange={setDate}
      theme="light"
      use24Hours={false}
    />
  );
}
